\section{Recap}

\begin{frame}
  \frametitle{Recap}

  \begin{itemize}
    \item \textbf{Go2Pins} is a tool that takes a \textit{Go} program and\
      links it to model-checkers.
    \item \textbf{Go2Pins} being at the early stages of development,\
      it lacks of features.
      \begin{itemize}
        \item \textbf{structs} are on their way to be supported.
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Recap}
  \framesubtitle{Objectives}

  \begin{center}
    \textbf{Go2Pins has proven himself!}\\
    \uncover<2->{But \texttt{struct}s are not handled in it.}\\
    \uncover<3->{Thus, the objective is to be able to fully handle \texttt{struct}.}
  \end{center}
\end{frame}
